56 // Create castesian coordinates neighbor lists
57 void Parallel::createNeighborLists(){
58 
59  // Count the number of subblocks per dimension
60  int subBlocksProd = 1;
61  for(int i = 0; i < 4; i++){
62  m_subBlocks[i] = m_latticeFullSize[i] / m_latticeSubSize[i];
63  subBlocksProd *= m_subBlocks[i];
64  }
65 
66  // Check if there are enough processors to fill the lattice
67  if(subBlocksProd > m_numProcs && m_rank == 0){
68  printf("ERROR: Too few processors given for specified lattice and sublattice sizes\n");
69  MPI_Abort(MPI_COMM_WORLD, 0);
70  } else {
71  m_activeProcs = subBlocksProd;
72  }
73 
74  // MPI_Cart required variables
75  std::array<int, 4> period = {1,1,1,1}; // for periodic boundaries
76  int reorder = 0; // no new rank naming
77 
78  // Create coordinates map
79  MPI_Cart_create(MPI_COMM_WORLD, 4, m_subBlocks.data(), period.data(), reorder, &m_cartCoords);
80 
81  // Create neighbor lists and intercommunicator
82  if(m_rank + 1 <= m_activeProcs){
83  m_isActive = true;
84  MPI_Cart_coords(m_cartCoords, m_rank, 4, m_rankCoord.data());
85  // Assign neighbors along all directions
86  for(int i = 0; i < 4; i++)
87  assignNeighbor(i);
88 
89  for(int i = 0; i < 4; i++)
90  for(int j = 0; j < 4; j++)
91  assignSecondNeighbor(i, j);
92  } else {
93  // Set extra processors to inactive
94  m_isActive = false;
95  }
96 
97  // Create intracommunicator
98  int active = m_isActive == true ? 1 : 0;
99  MPI_Comm_split(MPI_COMM_WORLD, active, 1, &m_cartCoordComm);
100 }
